    Навигация:
pwd -- Показывает текущую директорию, в которой находимся
ls -- Посмотреть все доступные файлы и папки в текущей директории
cd .. -- Выйти на уровень выше
cd ‘название папки’ -- перейти\открыть ‘название-папки’
clear -- Очистить терминал

touch ‘название файла’ -- создать файл в текущей папке

git clone https://посилання_на_репозиторій -- клонуєте репозиторій  
git branch -- Показує список гілок в проекті
git branch ‘назва-гілки’ -- Створює нову гілку
git branch -d ‘назва-гілки’ -- видалення гілки ‘назва-гілки’ локально

git checkout ‘назва-гілки’ -- перехід на гілку ‘назва-гілки’
git checkout -b ‘назва-нової-гілки’ -- створює нову гілку і переходить в неї

git pull origin ‘назва-гілки’ -- стягує останні зміни гілки ‘назва-гілки’
git add . -- зберігіємо зміни в файлах

git commit -m “назва-коміта” -- записує зміни до локальної гілки
git commit -am "назва коміта" -- зберігає зміни та записує зміни

git push origin ‘назва-гілки’ -- публікує ваші зміни на гілці віддаленого репозиторію
git push origin --delete ‘назва-гілки’ -- видаляє гілки з сайту GitHub

git stash -- зберігаємо не збережені зміни в файлах і кладемо їх в буфер обміну
git stash apply -- вставляємо збережені зміни з буфера обміну

git merge ‘назва-гілки’ -- залиття коду з гілки ‘назва-гілки’ на поточну гілку
git merge --abort -- Відмінити злиття гілок

git fetch  -- стягує всі зміни з віддаленого репозиторію
git log -- Побачити останні комміти
